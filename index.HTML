            const navItems = document.querySelectorAll('.nav-item');


            searchInput.addEventListener('input', (e) => {

                const term = e.target.value.toLowerCase();


                navSections.forEach(section => {

                    const items = section.querySelectorAll('.nav-item');

                    let hasVisibleChild = false;


                    items.forEach(item => {

                        const text = item.textContent.toLowerCase();

                        if (text.includes(term)) {

                            item.style.display = 'block';

                            hasVisibleChild = true;

                        } else {

                            item.style.display = 'none';

                        }

                    });


                    // Développer automatiquement la section si elle a des correspondances et que la recherche est active

                    if (term.length > 0) {

                        if (hasVisibleChild) {

                            section.style.display = 'block';

                            section.classList.add('expanded');

                        } else {

                            section.style.display = 'none';

                        }

                    } else {

                        // Réinitialiser quand la recherche est effacée

                        items.forEach(item => item.style.display = 'block');

                        // Ré-appliquer l'état localStorage

                        const currentState = JSON.parse(localStorage.getItem('gfxBibleNavState')) || {};

                        if (currentState[groupId] !== undefined) {

                            if (!currentState[groupId]) section.classList.remove('expanded');

                        }

                    }

                });

            });
